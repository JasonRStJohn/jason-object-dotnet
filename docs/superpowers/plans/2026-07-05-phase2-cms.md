# Phase 2 — CMS, Admin Seeding & Public Post Views Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close the content loop — an admin can create/edit/delete posts in a protected CMS, and anonymous visitors can read published posts at `/posts` and `/posts/{slug}`.

**Architecture:** Admin seeding at startup removes the need for open registration. A concrete `PostService` wraps EF Core CRUD. CMS pages are Interactive Server Blazor components under `/admin/posts`. Public post pages are SSR Blazor for SEO and fast load.

**Tech Stack:** .NET 9, Blazor Web App (SSR + InteractiveServer), EF Core 9 + SQL Server, ASP.NET Core Identity, xUnit + FluentAssertions + EF Core InMemory (tests)

## Global Constraints

- Target framework: `net9.0`
- All Blazor CMS pages: `@attribute [Authorize]` + `@rendermode InteractiveServer`
- All public Blazor pages: no `@rendermode` (SSR)
- No migrations needed — Post schema is complete from Phase 1
- `dotnet` is at `$HOME/.dotnet/dotnet`; always export `PATH="$HOME/.dotnet/tools:$HOME/.dotnet:$PATH"` and `DOTNET_ROOT="$HOME/.dotnet"` before running dotnet commands
- Run tests with: `dotnet test tests/MeDotNet.Tests/ --nologo --logger "console;verbosity=minimal"`

```csharp
// Seed admin user on first run
using (var scope = app.Services.CreateScope())
{
    var userManager = scope.ServiceProvider.GetRequiredService<UserManager<ApplicationUser>>();
    var logger = scope.ServiceProvider.GetRequiredService<ILogger<Program>>();
    var adminEmail = app.Configuration["ADMIN_EMAIL"];
    var adminPassword = app.Configuration["ADMIN_PASSWORD"];

    if (adminEmail is not null && adminPassword is not null)
    {
        try
        {
            var existing = await userManager.FindByEmailAsync(adminEmail);
            if (existing is null)
            {
                var user = new ApplicationUser { UserName = adminEmail, Email = adminEmail };
                var result = await userManager.CreateAsync(user, adminPassword);
                if (!result.Succeeded)
                    logger.LogError("Admin seed failed: {Errors}",
                        string.Join(", ", result.Errors.Select(e => e.Description)));
                else
                    logger.LogInformation("Admin user seeded: {Email}", adminEmail);
            }
        }
        catch (Exception ex)
        {
            logger.LogError(ex, "Admin seeding failed");
        }
    }
}
```

```csharp
using MeDotNet.Data;
using MeDotNet.Models;
using Microsoft.EntityFrameworkCore;

namespace MeDotNet.Services.Posts;

public class PostService(AppDbContext db)
{
    public Task<List<Post>> GetAllAsync() =>
        db.Posts.OrderByDescending(p => p.CreatedAt).ToListAsync();

    public Task<List<Post>> GetPublishedAsync() =>
        db.Posts.Where(p => p.PublishedAt != null)
                .OrderByDescending(p => p.CreatedAt)
                .ToListAsync();

    public Task<Post?> GetByIdAsync(int id) =>
        db.Posts.FirstOrDefaultAsync(p => p.Id == id);

    public Task<Post?> GetBySlugAsync(string slug) =>
        db.Posts.FirstOrDefaultAsync(p => p.Slug == slug && p.PublishedAt != null);

    public async Task CreateAsync(Post post)
    {
        db.Posts.Add(post);
        await db.SaveChangesAsync();
    }

    public async Task UpdateAsync(Post post)
    {
        db.Posts.Update(post);
        await db.SaveChangesAsync();
    }

    public async Task DeleteAsync(int id)
    {
        var post = await db.Posts.FindAsync(id);
        if (post is not null)
        {
            db.Posts.Remove(post);
            await db.SaveChangesAsync();
        }
    }
}
```

- [ ] **Step 1: Create PostEdit.razor**

Create `src/MeDotNet/Components/Pages/Admin/PostEdit.razor`:

```razor
@page "/admin/posts/new"
@page "/admin/posts/{Id:int}"
@attribute [Authorize]
@rendermode InteractiveServer
@inject PostService PostService
@inject NavigationManager Nav
@inject AuthenticationStateProvider AuthStateProvider

<PageTitle>@(_isNew ? "New Post" : "Edit Post") — Admin</PageTitle>
<h1>@(_isNew ? "New Post" : "Edit Post")</h1>

@if (_error is not null)
{
    <p style="color:red">@_error</p>
}

<div>
    <label>Title</label><br />
    <input type="text" value="@_title" @oninput="OnTitleInput" style="width:100%" />
</div>
<br />
<div>
    <label>Slug</label><br />
    <input type="text" value="@_slug" @oninput="OnSlugInput" style="width:100%" />
</div>
<br />
<div>
    <label>Body</label><br />
    <textarea rows="15" style="width:100%" @bind="_body"></textarea>
</div>
<br />
<div>
    <label>
        <input type="checkbox" @bind="_published" /> Published
    </label>
</div>
<br />
<button @onclick="SaveAsync">Save</button>
&nbsp;
<a href="/admin/posts">Cancel</a>

@code {
    [Parameter] public int? Id { get; set; }

    private string _title = "";
    private string _slug = "";
    private string _body = "";
    private bool _published;
    private bool _slugManuallyEdited;
    private string? _error;
    private bool _isNew => Id is null;
    private Post? _existing;

    protected override async Task OnInitializedAsync()
    {
        if (Id is not null)
        {
            _existing = await PostService.GetByIdAsync(Id.Value);
            if (_existing is not null)
            {
                _title = _existing.Title;
                _slug = _existing.Slug;
                _body = _existing.Body;
                _published = _existing.PublishedAt.HasValue;
                _slugManuallyEdited = true; // Don't overwrite slug when editing existing post
            }
        }
    }

    private void OnTitleInput(ChangeEventArgs e)
    {
        _title = e.Value?.ToString() ?? "";
        if (!_slugManuallyEdited)
            _slug = GenerateSlug(_title);
    }

    private void OnSlugInput(ChangeEventArgs e)
    {
        _slug = e.Value?.ToString() ?? "";
        _slugManuallyEdited = true;
    }

    private static string GenerateSlug(string title) =>
        System.Text.RegularExpressions.Regex.Replace(
            title.ToLowerInvariant().Replace(' ', '-'),
            @"[^a-z0-9-]", "").Trim('-');

    private async Task SaveAsync()
    {
        if (string.IsNullOrWhiteSpace(_title)) { _error = "Title is required."; return; }
        if (string.IsNullOrWhiteSpace(_slug))  { _error = "Slug is required."; return; }
        if (string.IsNullOrWhiteSpace(_body))  { _error = "Body is required."; return; }

        var authState = await AuthStateProvider.GetAuthenticationStateAsync();
        var userId = authState.User.FindFirst(System.Security.Claims.ClaimTypes.NameIdentifier)?.Value ?? "";

        if (_isNew)
        {
            var post = new Post
            {
                Title = _title,
                Slug = _slug,
                Body = _body,
                AuthorId = userId,
                PublishedAt = _published ? DateTime.UtcNow : null
            };
            await PostService.CreateAsync(post);
        }
        else if (_existing is not null)
        {
            _existing.Title = _title;
            _existing.Slug = _slug;
            _existing.Body = _body;
            _existing.PublishedAt = _published
                ? (_existing.PublishedAt ?? DateTime.UtcNow)  // Preserve original publish time if re-publishing
                : null;
            await PostService.UpdateAsync(_existing);
        }

        Nav.NavigateTo("/admin/posts");
    }
}
```
